Table WaferType {
  id integer [primary key, increment]
  name string
  engineeringRun engineeringRun
  foundry foundry
  technology technology
  imageBase64String string
  waferMap JSON
}

Table Wafer {
  id integer [primary key, increment]
  serialNumber string [unique]
  batchNumber integer
  thinningDate date
  dicingDate date
  productionDate date
  waferTypeId integer [ref: > WaferType.id]
}

Table Asic {
  id integer [primary key, increment]
  waferId integer [ref: > Wafer.id]
  serialNumber string [unique]
  familyType familyType
  waferMapPosition string
}

Table waferSubMap {
  waferTypeId integer [ref: > WaferType.id]
  orientation orientation
  submap JSON
}

Table wpMachineInfo {
  id integer [primary key, increment]
  name string [unique]
  hostname string
  connectionType connectionType
  connectionPort integer
  generalLocation generalLocation
  software software
  softwareVersion string
  vendor vendor
  probeCardInfo integer [ref: > probeCardInfo.id]
}

Table probeCardInfo {
  id integer [primary key, increment]
  serialNumber string [unique]
  vendor vendorPc
  name name
  model model
  version integer
  dateOfArrival date
  location location
  type type
  vendorCleaningInterval integer
}

Table probeCardFamilyType {
  probeCardInfoId integer [ref: > probeCardInfo.id]
  asicFamilyType asicFamilyType [ref: > Asic.familyType]
}

Table projectName {
  id integer [primary key, increment]
  wpMachineInfoId integer [ref: > wpMachineInfo.id]
  waferTypeId integer [ref: > WaferType.id]
  asicFamilyType asicFamilyType [ref: > Asic.familyType]
  orientation orientation
  name string [unique]
  alignmentDie string
  homeDie string
}

Table probeCardMaintenance {
  id integer [primary key, increment]
  probeCardInfoId integer [ref: > probeCardInfo.id]
  cleaningDate date
  totNumContacts integer
  numContactsSinceLastCleaning integer
  numContactsDuringCleaning integer
  cleaningOverdrive integer
}

Table asicProbingInfo {
  id integer [primary key, increment]
  asicId integer [ref: > Asic.id]
  numContacts integer
  mechanicalQuality mechanicalQuality
  dateAndTime date
}

Table waferLoadedInMachine {
  id integer [primary key, increment]
  machineId integer [ref: > wpMachineInfo.id]
  waferSerialNumber string [ref: > Wafer.serialNumber]
  date date
  userName string
  action action
}

Table asicConfiguration {
  id integer [primary key, increment]
  wpMachineInfoId integer [ref: > wpMachineInfo.id]
  versionId integer [ref: > probeCardInfo.version]
  testingAllowed boolean
}

Table wpConfiguration {
  id integer [primary key, increment]
  wpMachineInfoId integer [ref: > wpMachineInfo.id]
  versionId integer [ref: > probeCardInfo.version]
  orientation orientation
}

Table probeCardConfiguration {
  id integer [primary key, increment]
  probeCardInfoId integer [ref: > probeCardInfo.id]
  versionId integer [ref: > probeCardInfo.version]
  cleaningInterval integer
}
