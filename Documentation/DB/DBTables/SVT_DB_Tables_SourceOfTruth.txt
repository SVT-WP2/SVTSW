Table WaferType {
  id integer [primary key, increment]
  name string
  engineeringrun engineeringrun
  foundry foundry
  technology technology
  imagebase64string string
  wafermap JSON
}

Table Wafer {
  id integer [primary key, increment]
  serialnumber string [unique]
  batchnumber integer
  thinningdate date
  dicingdate date
  productiondate date
  wafertypeid integer [ref: > WaferType.id]
}

Table Asic {
  id integer [primary key, increment]
  waferid integer [ref: > Wafer.id]
  serialnumber string [unique]
  familytype familytype
  wafermapposition string
}

Table WaferSubMap {
  wafertypeid integer [ref: > WaferType.id]
  orientation orientation
  submap JSON
}

Table WpMachine {
  id integer [primary key, increment]
  name string [unique]
  hostname string
  connectiontype connectiontype
  connectionport integer
  generallocation generallocation
  software software
  softwareversion string
  vendor vendor
  probecardid integer [ref: > WpProbeCard.id]
}

Table WpProbeCard {
  id integer [primary key, increment]
  serialnumber string [unique]
  vendor vendorpc
  name name
  model model
  version integer
  arrivaldate date
  location location
  type type
  vendorcleaninginterval integer
}

Table WpProbeCardFamilyType {

}

Table WpProject {
  id integer [primary key, increment]

  orientation orientation
