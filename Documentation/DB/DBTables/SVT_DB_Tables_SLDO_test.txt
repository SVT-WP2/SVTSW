Table WaferType {
  id integer [primary key, increment]
  name string
  engineeringRun engineeringRun
  foundry foundryName
  technology waferTech
  waferMap JSON
}

Table Wafer {
  id integer [primary key, increment]
  serialNumber string [unique]
  batchNumber integer
  thinningDate date
  dicingDate date
  productionDate date
  waferTypeId integer [ref: > WaferType.id]
}

Table WaferLocation {
  waferId integer [ref: > Wafer.id, not null]
  generalLocation wpGeneralLocation
  creationTime timestamp [default: `CURRENT_TIMESTAMP`]
  username string
  description string
}

Table Asic {
  id integer [primary key, increment]
  waferId integer [ref: > Wafer.id]
  serialNumber string [unique]
  familyType asicFamilyType
  waferMapPosition string
  quality asicQuality
}

Table AncillaryAsic {
  asicId integer [ref: > Asic.id]
  ancAsicType ancAsicType
  location generalLocation
  arrivalDate date
}

Table TestSetup {
  ancAsicTestId integer [ref: > AncAsicTest.id]
  equipment equipmentName
  eqipmentDetails JSON
}


Table TestConfiguration {
  id integer [primary key]
  name testTypes
  mode modes
  voltageRampRates float
  loadCapacitance loadCap
  loadCurrent loadCurrent
  temperature temparature
}

Table AncAsicTest {
  id integer [primary key]
  name testTypes
  asicId integer [ref: > Asic.id]
  configId integer [ref: > TestConfiguration.id]
  testValues JSON
}
