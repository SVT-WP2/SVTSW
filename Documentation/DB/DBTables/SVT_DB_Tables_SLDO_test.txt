Table WaferType {
  id integer [primary key, increment]
  name string
  engineeringRun engineeringRun
  foundry foundryName
  technology waferTech
  waferMap JSON
}

Table Wafer {
  id integer [primary key, increment]
  serialNumber string [unique]
  batchNumber integer
  thinningDate date
  dicingDate date
  productionDate date
  waferTypeId integer [ref: > WaferType.id]
}

Table WaferLocation {
  waferId integer [ref: > Wafer.id, not null]
  generalLocation wpGeneralLocation
  creationTime timestamp [default: `CURRENT_TIMESTAMP`]
  username string
  description string
}

Table Asic {
  id integer [primary key, increment]
  waferId integer [ref: > Wafer.id]
  serialNumber string [unique]
  familyType asicFamilyType
  waferMapPosition string
  quality asicQuality
}

Table TestConfiguration {
  id integer [primary key]
  name tests
  mode Modes
  voltageRampRates float
  loadCapacitance Load_cap
  loadCurrent Load_current
  temperature Temparature
}

Table SLDO_Test {
  id integer [primary key]
  name tests
  asicId integer [ref: > Asic.id]
  configId integer [ref: > TestConfiguration.id]
  testValues JSON
}

Table Power_Ramp_up  {
  id integer [primary key]
  asicId integer [ref: > Asic.id]
  configId integer [ref: > TestConfiguration.id]
  V_in_target float
  I_in_limit float
  V_out float
}

Table PSRR  {

  I_in_limit float
  signal_amp float
  signal_freq float
  V_out_amp float
  PSRR float

Table Power_ramp_rate{

  V_in_target float
  I_in_limit float
  T_ramp ramp_time
  V_out float
}

Table DAC_scan{
  Test_ID integer [ref: > TestConfiguration.id]
  V_in_target float
  I_in_limit float
  DAC_code integer
  V_out float
}

Table Over_current{
  Test_ID integer [ref: > TestConfiguration.id]
  V_in_target float
  I_in_limit float
  I_excess float
  V_out float
  I_out float
  I_in float
}

Table Irradiation{

  I_in_limit float
  Iraddiation_dose float
  V_out float
  I_out float
  I_in float
